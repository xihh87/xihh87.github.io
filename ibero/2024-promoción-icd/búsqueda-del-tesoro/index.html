        self.first_names = ["John", "Emily", "Michael", "Sophia", "William", "Olivia"]
        self.last_names = ["Smith", "Johnson", "Williams", "Brown", "Jones", "Miller"]
        self.blood_types = ["A+", "A-", "B+", "B-", "AB+", "AB-", "O+", "O-"]
        self.medications = ["Aspirin", "Paracetamol", "Lisinopril", "Atorvastatin", "Levothyroxine"]
        self.allergies = ["Pollen", "Penicillin", "Peanuts", "Shellfish", "Dust"]
        self.surgeries = ["Appendectomy", "Cataract surgery", "Knee replacement", "Gallbladder removal"]
        self.coagulation_history = ["Normal", "Abnormal"]
        self.risk_factors = ["Smoking", "Alcohol consumption", "Obesity", "Hypertension"]

    def generate_personal_info(self):
        return {
            "name": f"{random.choice(self.first_names)} {random.choice(self.last_names)}",
            "date_of_birth": f"{random.randint(1, 28)}/{random.randint(1, 12)}/{random.randint(1950, 2000)}",
            "emergency_contact": f"{random.choice(self.first_names)} {random.choice(self.last_names)} - {random.randint(1000000000, 9999999999)}"
        }

    def generate_blood_type(self):
        return random.choice(self.blood_types)

    def generate_prescription_medication(self):
        return random.choice(self.medications)

    def generate_allergy(self):
        return random.choice(self.allergies)

    def generate_surgery_history(self):
        return random.choice(self.surgeries)

    def generate_coagulation_history(self):
        return random.choice(self.coagulation_history)

    def generate_risk_factor(self):
        return random.choice(self.risk_factors)
